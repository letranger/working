14: def read_img(img_path, resize=(299,299)):
15:     img_string = tf.io.read_file(img_path)  # 讀取檔案
16:     img_decode = tf.image.decode_image(img_string)  # 將檔案以影像格式來解碼
17:     img_decode = tf.image.resize(img_decode, resize)  # 將影像resize到網路輸入大小
18:     # 將影像格式增加到4維(batch, height, width, channels)，模型預測要求格式
19:     img_decode = tf.expand_dims(img_decode, axis=0)
20:     return img_decode
21: 
22: img = preprocess_input(img)  # 影像前處理
23: preds = model.predict(img)  # 預測圖片
24: print("Predicted:", decode_predictions(preds, top=3)[0])  # 輸出預測最高的三個類別
